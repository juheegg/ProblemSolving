int M, N, L
int gun[M], animal[N][2]

sort(gun)

int answer = 0
for(int i: 0~N-1)
	if(L >= animal[i][1])
		int min = animal[i][0] - (L - animal[i][1])
		int idx = lowerBound(min>=0?min:0)
		if(gun[idx]<= animal[i][0] + (L - animal[i][1]))
			answer++
	else
		continue
